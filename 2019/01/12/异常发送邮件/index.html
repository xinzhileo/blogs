public class DefaultHttpExceptionHandler implements HttpExceptionHandler {
    private static final Logger logger = LoggerFactory.getLogger(DefaultHttpExceptionHandler.class);
    private static final int EXCEPTION_SWITH = PropertyUtils.getInt("exception.remind.switch", 1);
    private static final String EXCEPTION_PROJECT = PropertyUtils.getString("exception.remind.project", "not set");
    private static final String EXCEPTION_PROFILE = PropertyUtils.getString("exception.remind.profile", "prod");
    private static final String LOCAL_IP = StrUtils.getLocalHostAddress();

    public DefaultHttpExceptionHandler() {
    }

    public void exceptionHandle(HttpRequest request, HttpResponse response, Exception exception) {
        try {
            SecurityUtils.getSubject().logout();
        } catch (Exception var5) {
            ;
        }

        if (!(exception instanceof AuthenticationException) && !(exception instanceof UnauthenticatedException) && !(exception.getCause() instanceof AuthenticationException) && !(exception.getCause() instanceof UnauthenticatedException)) {
            if (!(exception instanceof UnauthorizedException) && !(exception.getCause() instanceof UnauthorizedException)) {
                if (exception instanceof FinhubException) {
                    ResponseResultUtils.fail(response, (FinhubException)exception);
                } else if (exception.getCause() != null && exception.getCause() instanceof FinhubException) {
                    ResponseResultUtils.fail(response, (FinhubException)exception.getCause());
                } else if (exception.getCause() != null && exception.getCause() instanceof ValidateException) {
                    ResponseResultUtils.fail(response, new FinhubException(600, exception.getCause().getMessage()));
                } else {
                    FinhubLogger.error("内部错误", new Object[]{exception});
                    logger.error(ExceptionUtils.getErrorMessageWithNestedException(exception), exception);
                    this.exceptionRemind(request, exception);
                    ResponseResultUtils.exception(response);
                }
            } else {
                ResponseResultUtils.fail(response, new FinhubException(403));
            }
        } else {
            ResponseResultUtils.fail(response, new FinhubException(401));
        }

    }

    protected void exceptionRemind(HttpRequest request, Exception exception) {
        try {
            if (EXCEPTION_SWITH == 0) {
                return;
            }

            String url = "http://harmony-dev.fenbeijinfu.com/harmony/mail/inner";
            Map<String, Object> content = Maps.newLinkedHashMap();
            content.put("toList", Lists.newArrayList(new String[]{"rd@fenbeitong.com"}));
            content.put("subject", StrUtils.formatString("项目【{0}】【{1}】【{2}】发生内部错误", new Object[]{EXCEPTION_PROJECT, EXCEPTION_PROFILE, LOCAL_IP}));
            StringBuffer text = new StringBuffer();
            text.append("requestId：").append(request.getRequestId()).append("\n");
            text.append("requestURI：").append(request.getUri()).append("\n");
            if (exception.getCause() != null && exception.getCause() instanceof Exception) {
                exception = (Exception)exception.getCause();
            }

            text.append("exception：").append(ExceptionUtils.getStackTraceAsString(exception));
            content.put("text", text.toString());
            HttpClientUtils.postBody(url, JSON.toJSONString(content));
        } catch (Exception var6) {
            logger.error("发送异常邮件异常：" + var6.getMessage(), var6);
        }

    }
}
